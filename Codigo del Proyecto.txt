    void mostrardatos(String valor){
        //crear una variable tipo Default
    DefaultTableModel modelo=new DefaultTableModel();
    //lo que va contener la tabla
    modelo.addColumn("Nombre");
    modelo.addColumn("Apellido");
    modelo.addColumn("Usuario");
    modelo.addColumn("Correo");
    modelo.addColumn("Contraseña");
    //imprimimos en la tabla 
    tbusuarios.setModel(modelo);
    //recibiendo danto txtbuscar
    String sql="";
      if(valor.equals(""))
        {
         sql="SELECT *FROM usuarios";
        }
      else{
            sql="SELECT *FROM usuarios WHERE Usuario='"+valor+"'";
          }
    //vector para los datos
    String []datos = new String [5];
        try {
            //visualizar los datos
            Statement st= cn.createStatement();
            //almacena los datos de la consulta
            ResultSet rs = st.executeQuery(sql);
            while(rs.next()){
               datos[0]=rs.getString(1);
               datos[1]=rs.getString(2);
               datos[2]=rs.getString(3);
               datos[3]=rs.getString(4);
               datos[4]=rs.getString(5);
               //agregamos las filas
               modelo.addRow(datos);
            }
            //mostrar datos capturados
            tbusuarios.setModel(modelo);
        } catch (SQLException ex) {
            Logger.getLogger(MostrarUsuarios.class.getName()).log(Level.SEVERE, null, ex);
        }
    
    }

        void mostrardatos(String valor){
        //crear una variable tipo Default
    DefaultTableModel modelo=new DefaultTableModel();
    //lo que va contener la tabla
    modelo.addColumn("Nombre");
    modelo.addColumn("Apellido");
    modelo.addColumn("Usuario");
    modelo.addColumn("Correo");
    modelo.addColumn("Contraseña");
    //imprimimos en la tabla 
    tbusuarios.setModel(modelo);
    //recibiendo danto txtbuscar
    String sql="";
      if(valor.equals(""))
        {
         sql="SELECT *FROM usuarios";
        }
      else{
            sql="SELECT *FROM usuarios WHERE Usuario='"+valor+"'";
          }
    //vector para los datos
    String []datos = new String [5];
        try {
            //visualizar los datos
            Statement st= cn.createStatement();
            //almacena los datos de la consulta
            ResultSet rs = st.executeQuery(sql);
            while(rs.next()){
               datos[0]=rs.getString(1);
               datos[1]=rs.getString(2);
               datos[2]=rs.getString(3);
               datos[3]=rs.getString(4);
               datos[4]=rs.getString(5);
               //agregamos las filas
               modelo.addRow(datos);
            }
            //mostrar datos capturados
            tbusuarios.setModel(modelo);
        } catch (SQLException ex) {
            Logger.getLogger(MostrarUsuarios.class.getName()).log(Level.SEVERE, null, ex);
        }
    
    }

         try {
            PreparedStatement pst =cn.prepareStatement("INSERT INTO usuarios (Nombre,Apellido,Usuario,Correo,Contraseña) VALUES(?,?,?,?,?)");
            pst.setString(1, txtnom.getText());

            pst.setString(4, txtcor.getText());
            pst.setString(5, txtcon.getText());

            pst.executeUpdate();

            }
    }

    private void jMenuItem1ActionPerformed(java.awt.event.ActionEvent evt) {                                           
        // modificar
        int fila=tbusuarios.getSelectedRow();
        if(fila >= 0){
            txtnom.setText(tbusuarios.getValueAt(fila, 0).toString());
            txtape.setText(tbusuarios.getValueAt(fila, 1).toString());
            txtusu.setText(tbusuarios.getValueAt(fila, 2).toString());
            txtcor.setText(tbusuarios.getValueAt(fila, 3).toString());
            txtcon.setText(tbusuarios.getValueAt(fila, 4).toString());
         
            
        }
        else{
           JOptionPane.showMessageDialog(null,"NO SELECCIONO FILA");
        }
    }                                          

    private void btnmodificarActionPerformed(java.awt.event.ActionEvent evt) {                                             
             //modificar
        try {
          
            PreparedStatement pst= cn.prepareStatement("UPDATE usuarios SET Nombre='"+
                    txtnom.getText()+"',Apellido='"+txtape.getText()+"',Contraseña='"+

            //ejecutamos la sentencia
            pst.executeUpdate();
            mostrardatos("");

        }
   
            
    }
